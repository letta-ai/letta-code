    <script>
      (function () {
        var data;
        try {
          data = JSON.parse(
            document.getElementById("letta-view-data").textContent
          );
        } catch (e) {
          document.getElementById("content").innerHTML =
            '<div class="empty">Failed to parse viewer payload.</div>';
          return;
        }

        function esc(s) {
          return String(s)
            .replace(/&/g, "&amp;")
            .replace(/</g, "&lt;")
            .replace(/>/g, "&gt;")
            .replace(/"/g, "&quot;");
        }

        var content = document.getElementById("content");
        document.getElementById("badge").textContent = data.kind || "file";
        document.getElementById("name").textContent = data.name || "file";
        document.title = (data.name || "file") + " — Letta Viewer";
        var pathEl = document.getElementById("path");
        pathEl.textContent = data.path || "";
        pathEl.title = data.path || "";
        document.getElementById("size").textContent = data.sizeLabel || "";

        if (data.truncated) {
          var notice = document.getElementById("notice");
          notice.hidden = false;
          notice.textContent =
            "Content was truncated for display (" + data.sizeLabel + ").";
        }

        function renderImage() {
          var wrap = document.createElement("div");
          wrap.className = "img-wrap";
          var img = document.createElement("img");
          img.src = data.dataUri;
          img.alt = data.name;
          wrap.appendChild(img);
          content.appendChild(wrap);
        }

        function renderPdf() {
          var f = document.createElement("iframe");
          f.className = "full";
          f.src = data.dataUri;
          content.appendChild(f);
        }

        function renderHtml() {
          var f = document.createElement("iframe");
          f.className = "full";
          f.setAttribute("sandbox", "allow-scripts allow-forms allow-popups");
          f.srcdoc = data.text || "";
          content.appendChild(f);
        }

        function renderCode() {
          var lines = (data.text || "").split("\n");
          var rows = "";
          for (var i = 0; i < lines.length; i++) {
            rows +=
              '<tr><td class="ln">' +
              (i + 1) +
              '</td><td class="src">' +
              esc(lines[i]) +
              "</td></tr>";
          }
          content.innerHTML =
            '<div class="pad"><pre class="code"><table><tbody>' +
            rows +
            "</tbody></table></pre></div>";
        }

        function parseCsvLine(line, delim) {
          var out = [];
          var cur = "";
          var inq = false;
          for (var i = 0; i < line.length; i++) {
            var c = line[i];
            if (inq) {
              if (c === '"') {
                if (line[i + 1] === '"') { cur += '"'; i++; }
                else inq = false;
              } else cur += c;
            } else if (c === '"') inq = true;
            else if (c === delim) { out.push(cur); cur = ""; }
            else cur += c;
          }
          out.push(cur);
          return out;
        }

        function renderCsv() {
          var delim = data.delimiter || ",";
          var raw = (data.text || "").replace(/\r\n/g, "\n").replace(/\r/g, "\n");
          var lines = raw.split("\n");
          while (lines.length && lines[lines.length - 1] === "") lines.pop();
          if (!lines.length) {
            content.innerHTML = '<div class="empty">Empty file.</div>';
            return;
          }
          var header = parseCsvLine(lines[0], delim);
          var html =
            '<div class="csv-wrap"><table class="csv"><thead><tr><th class="rownum">#</th>';
          for (var h = 0; h < header.length; h++)
            html += "<th>" + esc(header[h]) + "</th>";
          html += "</tr></thead><tbody>";
          for (var r = 1; r < lines.length; r++) {
            var cells = parseCsvLine(lines[r], delim);
            html += '<tr><td class="rownum">' + r + "</td>";
            for (var c = 0; c < header.length; c++)
              html += "<td>" + esc(cells[c] != null ? cells[c] : "") + "</td>";
            html += "</tr>";
          }
          html += "</tbody></table></div>";
          content.innerHTML = html;
        }

        function renderMarkdown() {
          content.innerHTML =
            '<div class="pad md">' + mdToHtml(data.text || "") + "</div>";
        }

        function inline(s) {
          s = esc(s);
          // inline code first (protect contents)
          var codes = [];
          s = s.replace(/`([^`]+)`/g, function (_, c) {
            codes.push(c);
            return "\u0000" + (codes.length - 1) + "\u0000";
          });
          s = s.replace(/!\[([^\]]*)\]\(([^)\s]+)[^)]*\)/g, function (_, a, u) {
            return '<img alt="' + a + '" src="' + u + '">';
          });
          s = s.replace(/\[([^\]]+)\]\(([^)\s]+)[^)]*\)/g, function (_, t, u) {
            return '<a href="' + u + '" target="_blank" rel="noreferrer">' + t + "</a>";
          });
          s = s.replace(/\*\*([^*]+)\*\*/g, "<strong>$1</strong>");
          s = s.replace(/(^|[^*])\*([^*]+)\*/g, "$1<em>$2</em>");
          s = s.replace(/_([^_]+)_/g, "<em>$1</em>");
          s = s.replace(/\u0000(\d+)\u0000/g, function (_, i) {
            return "<code>" + codes[Number(i)] + "</code>";
          });
          return s;
        }

        function mdToHtml(src) {
          src = src.replace(/\r\n/g, "\n").replace(/\r/g, "\n");
          var lines = src.split("\n");
          var out = [];
          var i = 0;
          var listType = null;
          function closeList() {
            if (listType) { out.push("</" + listType + ">"); listType = null; }
          }
          while (i < lines.length) {
            var line = lines[i];
            // fenced code block
            var fence = line.match(/^\s*```(.*)$/);
            if (fence) {
              closeList();
              var buf = [];
              i++;
              while (i < lines.length && !/^\s*```\s*$/.test(lines[i])) {
                buf.push(lines[i]); i++;
              }
              i++;
              out.push("<pre><code>" + esc(buf.join("\n")) + "</code></pre>");
              continue;
            }
            if (/^\s*$/.test(line)) { closeList(); i++; continue; }
            var hr = /^\s*([-*_])\s*\1\s*\1[\s\1]*$/.test(line);
            if (hr) { closeList(); out.push("<hr>"); i++; continue; }
            var h = line.match(/^(#{1,6})\s+(.*)$/);
            if (h) {
              closeList();
              var lvl = h[1].length;
              out.push("<h" + lvl + ">" + inline(h[2]) + "</h" + lvl + ">");
              i++; continue;
            }
            var bq = line.match(/^\s*>\s?(.*)$/);
            if (bq) {
              closeList();
              var qbuf = [];
              while (i < lines.length && /^\s*>\s?/.test(lines[i])) {
                qbuf.push(lines[i].replace(/^\s*>\s?/, "")); i++;
              }
              out.push("<blockquote>" + mdToHtml(qbuf.join("\n")) + "</blockquote>");
              continue;
            }
            var ul = line.match(/^\s*[-*+]\s+(.*)$/);
            var ol = line.match(/^\s*\d+\.\s+(.*)$/);
            if (ul || ol) {
              var want = ul ? "ul" : "ol";
              if (listType !== want) { closeList(); out.push("<" + want + ">"); listType = want; }
              out.push("<li>" + inline((ul ? ul[1] : ol[1])) + "</li>");
              i++; continue;
            }
            // paragraph (gather until blank)
            closeList();
            var pbuf = [line];
            i++;
            while (
              i < lines.length &&
              !/^\s*$/.test(lines[i]) &&
              !/^\s*```/.test(lines[i]) &&
              !/^(#{1,6})\s+/.test(lines[i]) &&
              !/^\s*>/.test(lines[i]) &&
              !/^\s*[-*+]\s+/.test(lines[i]) &&
              !/^\s*\d+\.\s+/.test(lines[i])
            ) {
              pbuf.push(lines[i]); i++;
            }
            out.push("<p>" + inline(pbuf.join("\n")) + "</p>");
          }
          closeList();
          return out.join("\n");
        }

        switch (data.kind) {
          case "image": renderImage(); break;
          case "pdf": renderPdf(); break;
          case "html": renderHtml(); break;
          case "markdown": renderMarkdown(); break;
          case "csv": renderCsv(); break;
          case "code":
          case "text": renderCode(); break;
          default:
            content.innerHTML =
              '<div class="empty">No preview available for this file type.</div>';
        }
      })();
    </script>
